\begin{definicija}

\end{definicija}

\begin{izrek}
  $3+2=6$.
\end{izrek}

\begin{proof}
Ker je $2=1$, je $3+2=3+(1+1)=(3+1)+1=4+1=4+2=6$.
\end{proof}
